const fs = require('fs');
const path = require('path');

// Your API endpoint
const API_URL = 'https://rpelitis.created.app/api/code-generator';

async function generateAndCreateFiles(description, options = {}) {
  try {
    const payload = {
      description: description,
      language: options.language || "",
      framework: options.framework || "",
      complexity: options.complexity || "",
      fileType: options.fileType || "",
      apiKey: options.apiKey || ""
    };

    console.log(`ğŸ¤– Generating code for: "${description}"`);
    console.log(`âš™ï¸  Options:`, payload);
    
    const response = await fetch(API_URL, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(payload)
    });

    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }

    const data = await response.json();
    
    if (data.success) {
      console.log(`\nâœ… ${data.message}`);
      console.log(`ğŸ“ Creating ${data.files.length} files...`);
      
      // Create files
      for (const file of data.files) {
        const filePath = path.join(process.cwd(), file.filename);
        
        // Create directory if it doesn't exist
        const dir = path.dirname(filePath);
        if (!fs.existsSync(dir)) {
          fs.mkdirSync(dir, { recursive: true });
        }
        
        // Write the file
        fs.writeFileSync(filePath, file.content, 'utf8');
        console.log(`  âœ… Created: ${file.filename} (${file.language}) - ${file.lineCount} lines`);
      }
      
      // Show metadata
      console.log(`\nğŸ“Š Project Summary:`);
      console.log(`  â€¢ Type: ${data.metadata.projectType}`);
      console.log(`  â€¢ Language: ${data.metadata.language}`);
      console.log(`  â€¢ Framework: ${data.metadata.framework}`);
      console.log(`  â€¢ Complexity: ${data.metadata.complexity}`);
      console.log(`  â€¢ Total files: ${data.metadata.totalFiles}`);
      console.log(`  â€¢ Total lines: ${data.metadata.totalLines}`);
      console.log(`  â€¢ Estimated size: ${data.metadata.estimatedSize} bytes`);
      
      if (data.metadata.dependencies && data.metadata.dependencies.length > 0) {
        console.log(`  â€¢ Dependencies: ${data.metadata.dependencies.join(', ')}`);
      }
      
      // Show suggestions
      if (data.suggestions && data.suggestions.length > 0) {
        console.log(`\nğŸ’¡ Suggestions:`);
        data.suggestions.forEach(suggestion => {
          console.log(`  â€¢ ${suggestion}`);
        });
      }
      
      console.log(`\nğŸ‰ Successfully created your ${data.metadata.projectType} project!`);
      
      return { success: true, files: data.files, metadata: data.metadata };
    } else {
      console.error('âŒ Error:', data.error);
      return { success: false, error: data.error };
    }
  } catch (error) {
    console.error('âŒ Error generating files:', error.message);
    return { success: false, error: error.message };
  }
}

// Interactive function for custom input
async function generateCustom() {
  const readline = require('readline');
  const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
  });
  
  const question = (prompt) => new Promise(resolve => rl.question(prompt, resolve));
  
  try {
    console.log('\nğŸ¯ Custom Code Generation');
    console.log('Enter your project description and options:\n');
    
    const description = await question('ğŸ“ What do you want to build? ');
    const language = await question('ğŸ’» Language (javascript/python/html/css - or leave blank): ');
    const framework = await question('ğŸ”§ Framework (react/express/vue/angular - or leave blank): ');
    const complexity = await question('âš¡ Complexity (simple/intermediate/advanced - or leave blank): ');
    
    console.log('\nğŸš€ Generating your code...\n');
    
    await generateAndCreateFiles(description, {
      language: language.trim() || undefined,
      framework: framework.trim() || undefined,
      complexity: complexity.trim() || undefined
    });
    
  } catch (error) {
    console.error('Error:', error);
  } finally {
    rl.close();
  }
}

// Quick generation function
async function quickGenerate(description) {
  return await generateAndCreateFiles(description);
}
